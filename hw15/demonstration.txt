>>>Mikrotik = Router("Mikrotik", 4)

Mikrotik router with 4 ports has been created.

# Мы создали объект Mikrotik класса Router с 4 портами.

>>>Mikrotik.add_ipv4("192.168.5.14/24", 'eth1')

192.168.5.14/24 is assigned to interface eth1

# Назначили IPv4 адресс на интерфейс eth1

>>>Mikrotik.show_ip_table()

Interface                IPv4 Address/Mask
----------------------------------------------------------
eth1                     192.168.5.14/24
eth2                     Not defined
eth3                     Not defined
eth4                     Not defined

# Вывели IP таблицу

>>>Mikrotik.show_ip_route()

Destination Network                      Gateway
----------------------------------------------------------
192.168.5.0/24                             192.168.5.14/24 eth1

# Вывели IP Route таблицу, где автоматический появилось запись о том, что у нас есть
# connected сеть через порт eth1, так как мы до этого присвоили этому порту IPv4 адрес.

>>>Mikrotik.add_route("172.16.0.0/16", "192.168.5.1/24")

Adding route to 172.16.0.0/16 via 192.168.5.1/24 - ok

# Успешно добавили маршрут к сети 172.16.0.0/16 через 192.168.5.1/24, так как у нас
# уже имеется маршрут к сети 192.168.5.0/24

>>>Mikrotik.show_ip_route()

Destination Network                      Gateway
----------------------------------------------------------
192.168.5.0/24                             192.168.5.14/24 eth1
172.16.0.0/16                             192.168.5.1/24

# Таблица маршрутизации автоматически обновилась
